学习通全自动刷课脚本

功能介绍：

自动识别视频课程
自动2倍速播放
自动静音、防止暂停
自动跳过章节测验、PPT页面
视频播完自动下一集
全程无需手动操作

学习通网页版

2:鼠标右键，然后在弹出的菜单里面点检查



3:在右边上面（有图标的什么元素，后面那个是控制台）（鼠标在上面呆久了应该会告诉你



4:将下面代码复制在控制台里面，最后按一个回车就可以了

(function(){

    let iframe1
    let iframe2
    let video
    let preventpausing
    let jumpexampage

    function getvideo(){
        //  iframe1 = document.querySelector("iframe");
        //  iframe2 = iframe1.contentDocument.querySelector("iframe.ans-insertvideo-online");//进入视频所在//
        //  video = iframe2.contentDocument.querySelector("video#video_html5_api");//抓住视频//



        
    try {
    iframe1=document.querySelector("iframe");
    try {
        iframe2=iframe1.contentDocument.querySelector("iframe.ans-insertvideo-online");//进入视频所在//
        try {
            video=iframe2.contentDocument.querySelector("video#video_html5_api");//抓住视频//
        } catch(e){goNextSection()
            video=null
        }
    } catch(e){goNextSection()
        video=null
    }
} catch(e){goNextSection()
    video=null
}

    }
    function play1(){
        if(video){
             video.playbackRate=2
             video.muted = true
             video.play()
             clearInterval(preventpausing);
             preventpausing=setInterval(function() {
                 if (video && video.paused) {
                     video.play();} }, 2000);

            video.onended = function() {
             // 3. 关掉上面那个自动播放定时器
             clearInterval(preventpausing);
             isvideopage()
                                        }

    
 
                }          
        }
    function goNextSection() {
    // 获取“下一节”按钮
    const nextBtn = document.querySelector("#prevNextFocusNext");

    // 如果找到，就点击切换
    if (nextBtn) {
        nextBtn.click();
        console.log("✅ 已切换到下一节");
    } else {
        console.log("❌ 未找到下一节按钮");
    }
    }


    function isvideopage(){
        goNextSection()
         jumpexampage=setTimeout(function(){
            getvideo()
            if (video){
                play1()
                return
            }
            isvideopage()
        }

        ,5000)
    }

    function run() {
        console.log('菩萨保佑，开始了');
        getvideo()
        play1()
        
    }



run()

})()
